# 动态规划

重要点：

  • 寻找状态转移方程
  • 建立合适的数据结构表（多为类数组结构）
  • 返回符合题意的结果

# 爬楼梯

经典问题

function climbStairs (n) {
    let dp1 = 0, dp2 = 1, res = 1
    for(let i = 1; i < n; i++) {
        dp1 = dp2
        dp2 = res
        res = dp1 + dp2
    }
    return res
}
